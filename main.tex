\end{equation*}

    \begin{align*}
        \Delta^kx(n)&=(E-I)^kx(n)\\
        &=\sum_{i=0}^k(-1)^i{k\choose i}E^{k-i}x(n)\\
        &=\sum_{i=0}^k(-1)^i{k\choose i}x(n+k-i)
    \end{align*}

    \begin{align*}
        E^k&=(\Delta + I)^k\\
        &=\sum_{i=0}^k{k \choose i} \Delta^{k-i}(I)^i

        E^kx(n) =\sum_{i=0}^k{k \choose i} \Delta^{k-i}x(n)\\
